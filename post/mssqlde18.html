មេរៀនទី៣០ ៖ (Data Manipulation Languages)DML Trigger

Syntax:

  Create Trigger Trigger_Name

  On TableName After/For/Instead of Insert/Delete/Update

  As

  Insert,Delete,Update Statement;

Example:

  ដំបូងយើងត្រូវបង្កើត Trigger មួយសម្រាប់ធ្វើការទៅកាន់ Table ណាមួយដែលអ្នកចង់ធ្វើការអ្វីមួយបន្ទាប់ពី Delete Data ចូលក្នុង Table នោះ ។ ចំណែកឯខាងក្រោមនេះវិញគឺយើងបង្កើត Trigger មួយឈ្មោះ T_for_delete ដែលនឹងធ្វើការ Delete ទិន្នន័យទៅលើ Table Persons ហើយយើងធ្វើការ Insert Record ដែលយើងបាន Delete ចូលក្នុង Table tblDelete ។

Ex1:

Create Trigger [dbo].[T_for_delete]

on [dbo].[tblPersons] for DELETE

  as

  Declare @ID int=0;

  Select @ID=id from DELETED;

  Insert into tblDELETED(Record_No,Dates) Values(@ID,GetDate())



សម្រាប់ “DELETED” គឺជាTemp Table ដែលវាមានតម្លៃនៅពេលដែល យើងលុប Records ណាមួយនៅក្នុង Table ដែលយើងបានបង្កើត Trigger នៅលើ Table នោះ ។ ហើយ INSERTED សម្រាប់ Table ដែលយើងបាន Insert ទៅកាន់ table នោះ ។ Update ???

Ex2:

Create Trigger [dbo].[T_for_update]

on [dbo].[tblPersons] for Update

  as

  Declare @ID int=0;

  Select @ID=id from DELETED;

  Insert into tblDELETED(Record_No,Dates) Values(@ID,GetDate())

  Select @ID=id from INSERTED;

  Insert into tblINSERTED(Record_No,Dates) Values(@ID,GetDate())

 Ex3:

Create Trigger [dbo].[T_for_update]

on [dbo].[tblPersons] Instead of Delete

  as

  Declare @ID int=0;

  Select @ID=id from DELETED;

  Update tblPersons Set Status=0 Where ID=@ID;


2. (Data Definition Languages)DDL Trigger : សម្រាប់ធ្វើការទៅលើការប្រែប្រួលទៅលើStructure នៅក្នុង Table/Procedure ឬ Object ណាមួយដូចជា៖ Create, Alter, Drop និង Grant, Revoke ។

  ខាងក្រោមនេះ ជាឧទាហរណ៍សម្រាប់ធ្វើការតាមដានទៅលើការលុប, បង្កើត, កែប្រែ Table ហើយទិន្នន័យទាំងនោះដាក់ចូលក្នុង Table DDLEvent ។

  ដំបូងយើងបង្កើត Table:

CREATE TABLE dbo.DDLEvents

(

    EventDate    DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,

    EventType    NVARCHAR(64),

    EventDDL     NVARCHAR(MAX),

    EventXML     XML,

    DatabaseName NVARCHAR(255),

    SchemaName   NVARCHAR(255),

    ObjectName   NVARCHAR(255),

    HostName     VARCHAR(64),

    IPAddress    VARCHAR(32),

    ProgramName  NVARCHAR(255),

    LoginName    NVARCHAR(255)

);

  បន្ទាប់មកយើងធ្វើការបង្កើត Trigger ទៅលើ Table:

Create TRIGGER DDLTrigger_Table

    ON DATABASE    FOR CREATE_Table, ALTER_Table, DROP_Table

AS

BEGIN

    DECLARE @EventData XML = EVENTDATA();

    DECLARE @ip VARCHAR(32) =

        (SELECT client_net_address FROM sys.dm_exec_connections WHERE session_id = @@SPID);

    INSERT DDLEvents(EventType,EventDDL,EventXML,DatabaseName,SchemaName,

        ObjectName,HostName,IPAddress,ProgramName,LoginName)

    SELECT

        @EventData.value('(/EVENT_INSTANCE/EventType)[1]',   'NVARCHAR(100)'),

        @EventData.value('(/EVENT_INSTANCE/TSQLCommand)[1]', 'NVARCHAR(MAX)'),

        @EventData,  DB_NAME(), @EventData.value('(/EVENT_INSTANCE/SchemaName)[1]',  'NVARCHAR(255)'),

        @EventData.value('(/EVENT_INSTANCE/ObjectName)[1]',  'NVARCHAR(255)'),

        HOST_NAME(), @ip, PROGRAM_NAME(), SUSER_SNAME();

END
